    "import gymnasium as gym\n",
    "import numpy as np\n",
    "import os\n",
    "from PIL import Image\n",
    "from collections import defaultdict\n",
    "from gymnasium.wrappers import RecordVideo\n",
    "\n",
    "def preprocess_obs(obs_rgb, size=(16, 16), threshold=100):\n",
    "    \"\"\"\n",
    "    obs_rgb: np.ndarray (H, W, 3), uint8\n",
    "    1) Grayscale\n",
    "    2) Resize to 'size'\n",
    "    3) Binarize with threshold\n",
    "    Returns a tuple of 0/1 ints length size[0]*size[1]\n",
    "    \"\"\"\n",
    "    img = Image.fromarray(obs_rgb)\n",
    "    img = img.convert(\"L\")  # grayscale\n",
    "    img = img.resize(size, Image.BILINEAR)\n",
    "    arr = np.array(img, dtype=np.uint8)\n",
    "    binarized = (arr > threshold).astype(np.uint8)\n",
    "    return tuple(int(v) for v in binarized.flatten())\n",
    "\n",
    "def q_values_for_state(Q, state, n_actions):\n",
    "    return np.array([Q[(state, a)] for a in range(n_actions)], dtype=np.float32)\n",
    "\n",
    "env = gym.make(\"ALE/Pong-v5\", render_mode=\"rgb_array\")  # avoid render_mode=\"human\" for speed\n",
    "\n",
    "video_dir = \"./videos\"\n",
    "os.makedirs(video_dir, exist_ok=True)\n",
    "\n",
    "env = RecordVideo(env, video_folder=video_dir, episode_trigger=lambda ep: True)\n",
    "\n",
    "n_actions = env.action_space.n\n",
    "print(\"Actions:\", n_actions)\n",
    "\n",
    "# Hyperparameters\n",
    "alpha = 0.1\n",
    "gamma = 0.99\n",
    "epsilon = 0.2\n",
    "epsilon_min = 0.01\n",
    "epsilon_decay = 0.995\n",
    "episodes = 1000\n",
    "max_steps = 10000\n",
    "\n",
    "Q = defaultdict(float)\n",
    "rng = np.random.default_rng(0)\n",
    "\n",
    "for ep in range(1, episodes + 1):\n",
    "    obs, info = env.reset()\n",
    "    state = preprocess_obs(obs, size=(16, 16), threshold=100)\n",
    "    done = False\n",
    "    total_reward = 0.0\n",
    "    steps = 0\n",
    "    frame_count = 0\n",
    "\n",
    "    while not done and steps < max_steps:\n",
    "        # Epsilon-greedy\n",
    "        if rng.random() < epsilon:\n",
    "            action = env.action_space.sample()\n",
    "        else:\n",
    "            q_vals = q_values_for_state(Q, state, n_actions)\n",
    "            action = int(np.argmax(q_vals))\n",
    "\n",
    "        next_obs, reward, terminated, truncated, info = env.step(action)\n",
    "        done = terminated or truncated\n",
    "        total_reward += reward\n",
    "        steps += 1\n",
    "\n",
    "        next_state = preprocess_obs(next_obs, size=(16, 16), threshold=100)\n",
    "        frame_count += 1\n",
    "        # Q-learning update\n",
    "        current_q = Q[(state, action)]\n",
    "        next_max_q = 0.0 if done else np.max(q_values_for_state(Q, next_state, n_actions))\n",
    "        target = reward + gamma * next_max_q\n",
    "        Q[(state, action)] = (1 - alpha) * current_q + alpha * target\n",
    "\n",
    "        state = next_state\n",
    "    print(\"Frames recorded:\", frame_count)\n",
    "    epsilon = max(epsilon_min, epsilon * epsilon_decay)\n",
    "    print(f\"Episode {ep:03d} | Return: {total_reward:.1f} | Epsilon: {epsilon:.3f} | frame: {frame_count}\")\n",
    "\n",
    "env.close()"
